# Modal

Modal component exposes `close` and `open` methods :

<p-modal-preview></p-modal-preview>

<Source language='tsx' dark format={true} code={`
import { Component, Element, h } from '@stencil/core'

@Component({
    tag: 'p-alert-preview',
})
export class Alert {
    @Element()
    public el: HTMLElement

    public openModal() {
        this.el.querySelector('p-modal')?.open()
    }

    public closeModal() {
        this.el.querySelector('p-modal')?.close()
    }

    render() {
        return (
            <div>
                <p-modal>
                    <h1>
                        Ciao fratello
                    </h1>
                    <p>click on button, cross or outside modal to close it.</p>
                    <br />
                    <p-button type="success" onClick={() => this.closeModal()}>
                        Close modal
                    </p-button>
                </p-modal>
                <p-button type="primary" onClick={() => { this.openModal() }}>
                    Open modal
                </p-button>
            </div>
        )
    }
}

`} />
